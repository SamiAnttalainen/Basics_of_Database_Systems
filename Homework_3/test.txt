SELECT 
    U.Username AS User,
    C1.Content AS Comment,
    C2.CommentID AS "Commented on"
FROM 
    User U
INNER JOIN 
    Comments C1 ON U.UserID = C1.UserID
INNER JOIN 
    Comments C2 ON C1.FK_CommentID = C2.CommentID
ORDER BY 
    User ASC, 
    "Commented on" DESC;
